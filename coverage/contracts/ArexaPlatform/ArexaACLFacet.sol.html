pragma solidity ^0.8.9;
 
import { LibAccessControl } from "../base/AccessControl/LibAccessControl.sol";
 
import { LibArexaConst } from "./LibArexaConst.sol";
 
import { CallProtection } from "../base/Shared/ProtectedCall.sol";
import { ModifierRole } from "../base/AccessControl/ModifierRole.sol";
import { ModifierPausable } from "../base/TargetedPausable/ModifierPausable.sol";
 
contract ArexaACLFacet is CallProtection, ModifierRole, ModifierPausable {
	bytes32 public constant AREXA_ADMIN_ROLE = LibArexaConst.AREXA_ADMIN_ROLE;
 
	function hasRole(bytes32 role, address account) external view virtual EprotectedCall returns (bool) {
		return LibAccessControl._hasRole(role, account);
	}
 
	function grantRole(bytes32 role, address account) external virtual EprotectedCall EonlyRoleAdmin(role) {
		_grantRole(role, account);
	}
 
	function revokeRole(bytes32 role, address account) external virtual EprotectedCall EonlyRoleAdmin(role) {
		require(!(role == AREXA_ADMIN_ROLE && account == LibAccessControl._owner()), "Role cannot revoke from owner!");
		_revokeRole(role, account);
	}
 
	function renounceRole(bytes32 role) external virtual EprotectedCall {
		_renounceRole(role);
	}
 
	function getRoleAdmin(bytes32 role) external view virtual EprotectedCall returns (bytes32) {
		return LibAccessControl._getRoleAdmin(role);
	}
 
	function setRoleAdmin(bytes32 role, bytes32 adminRole) external EprotectedCall EonlyOwner {
		_setRoleAdmin(role, adminRole);
	}
 
	function _setRoleAdmin(bytes32 role, bytes32 adminRole) internal virtual EprotectedCall {
		LibAccessControl._setRoleAdmin(role, adminRole);
	}
 
	function _grantRole(bytes32 role, address account) internal virtual EprotectedCall {
		LibAccessControl._grantRole(role, account);
	}
 
	function _revokeRole(bytes32 role, address account) internal virtual EprotectedCall {
		LibAccessControl._revokeRole(role, account);
	}
 
	function _renounceRole(bytes32 role) internal virtual EprotectedCall {
		LibAccessControl._renounceRole(role);
	}
}
